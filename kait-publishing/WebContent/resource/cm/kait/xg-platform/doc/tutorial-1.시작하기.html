1.시작하기(5분)

본 문서는 Xg Platform을 처음 접하는 사용자를 대상으로 작성되어 있다. 클라이언트에서의 개발 방법에 대해서만 가이드하며, 서버에 대한 사항은 다음 Server API Documentation을 참고한다.

준비 사항

사용자의 하기의 폴더 구조 및 서버측에 API 가 준비되어 있다고 가정한다. 실제 서버가 없이 동작가능 하지만, 본 문서에는 서버와 연동을 한다는 가정하에 진행한다.

본 예제는 본인이 원하는 웹 서버 또는 웹 어플리케이션 서버 위에서 구동하도록 한다.

  • Xg Platform 다운로드 링크를 통해 최신버전을 다운로드를 받는다.
  • API URL : /XG/test/select.jsp

작성할 어플리케이션

본 예제는 서버측에 API를 호출하여 그리드에 데이터를 표현하는 단순한 어플리케이션을 개발하는 목표로 작성한다. 데이터 바인딩을 추가로 확인하기 위하여 입력상자 컨트롤을 추가 및 바인딩을 하고, 그리드의 로우를 클릭했을 때 데이터 바인딩이 정상적으로 이루어지는 지 확인한다.

1. index.html 생성

기본 마크업을 작성한다. 스타일 및 스크립트의 로딩 순서에 주의하며 작성한다.

{@lang xml}<!DOCTYPE html>
<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <meta http-equiv="X-UA-Compatible" content="IE=edge">
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <title>Xg Platform Sample App</title>
        <link rel="stylesheet" type="text/css" media="all" href="xg-platform/lib/swidgets/styles/swidgets.base.css">
        <!--[if lte IE 8]>
        <script src="../../lib/es5-shim/es5-shim.min.js"></script>
        <![endif]-->        

        <!-- dependencies lib module -->
        <script src="xg-platform/lib/json3/lib/json3.min.js"></script>
        <script src="xg-platform/lib/jquery/dist/jquery.min.js"></script>

        <!-- swidgets module -->
        <script src="xg-platform/lib/swidgets/swidgets.js"></script>

        <!-- xg-platform module -->
        <script src="xg-platform/xg-all.js"></script>
    </head>
    <body></body>
</html>

2. 그리드를 적용할 엘리먼트 및 스크립트( main.js ) 코드 추가

엘리먼트와 스크립트를 body tag에 추가한다. 스크립트 코드는 DOM이 랜더링 된 후 수행할 수 있도록 body tag 마지막 전에 작성한다.

{@lang xml}    <body>
        <h1>시작하기(5분)</h1>
        <input id="xg-input" type="text">
        <div id="xg-grid"></div>
        <script src="main.js"></script>
    </body>

3. 스크립트 코드 작성 ( js/main.js )

Restful API 호출 및 그리드에 데이터를 바인딩할 스크립트 코드를 작성한다. 코드를 실행 하였을 때 그리드의 로우를 선택하면 해당 로우의 'EMP_NO' 컬럼의 값이 입력상자 컨트롤의 값에 반영이 되는지 확인한다.

// dom이 로드 된 후 수행한다.
$( document ).ready( function () {

    // 'ds'라는 아이디를 지닌 새로운 데이터 셋 인스턴스를 생성한다.
    var ds = new XgDataSet( 'ds' );

    // 'dt'라는 아이디를 지닌 새로운 데이터 테이블 인스턴스를 생성한다. 
    // 생성 시 조회에 요청할 API URL을 설정한다.
    var dt = new XgDataTable( 'dt', {
        url: {
            select: '/XG/jsp/select.jsp'
        }
    } );

    // 'da'라는 아이디를 지닌 새로운 데이터 어댑터 인스턴스를 생성한다.
    // 데이터 어댑터는 실제 서버와 통신을 관장할 통신 모듈이다.
    var da = new XgDataAdapter( 'da' );

    // SOCD를 설정한다. 이 예제의 서버측 코드에 정의된 데이터 셋 이름 'ds' 을 전달해야한다.
    da.setSOCD( '(O:ds=dt,I:ds=dt)' )

    // 데이터 어탭터가 'XML' 방식으로 통신하도록 설정한다.
    da.setDataType( 'XML' );

    // 데이터 테이블 및 어댑터를 데이터 셋에 등록한다.
    ds.addDataTable( dt );
    ds.setDataAdapter( da );

    // 서버에 데이터를 요청 및 데이터를 수신하고 콜백함수에서 그리드 및 입력상자 컨트롤을 초기화 시킨다.
    ds.select( 'dt', function () {

        $( '#xg-grid' ).xgGrid( {
            width: 500,
            height: 500,
            xgInitOption: {
                xgDataSet: ds,
                xgDataTable: dt
            },
            xgBindOption: {
                template: 'view'
            }
        } );

        $( '#xg-input' ).xgInput( {
            width: 400,
            height: 30,
            placeHolder: '선택된 행의 EMP_NO 값이 바인딩 됩니다.',
            xgInitOption: {
                xgDataSet: ds,
                xgDataTable: 'dt'
            },
            xgBindOption: {
                mode: 'read',
                bind: {
                    type: 'column',
                    column: 'EMP_NO'
                }
            }
        } );

    } );

} );
